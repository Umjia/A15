#include <stdio.h>
#include<stdbool.h>  
int main()
{
    int arr[10][10],i,j,n;
    bool count=true;
    printf("Enter  n:");
    scanf("%d",&n);
    
    for (i=0;i<n;i++)    
    {    
        for (j=0;j<n;j++)    
        {    
            printf("Enter a[%d][%d]: ",i,j);                
            scanf("%d",&arr[i][j]);    
        }    
    }    
    printf("\n printing the elements ....\n");     
    for(i=0;i<n;i++)    
    {    
        printf("\n");    
        for (j=0;j<n;j++)    
        {    
            printf("%d\t",arr[i][j]);    
        }    
    }
    
    j==0;
     for(int i=0;i<n;i++)
        {
            for(int j=0;j<n-1;j++)
            {
                if(arr[j][i] > arr[j+1][i])
                {
                    continue;
                }
                else
                {
                    count=false;
                }
            }
        }
    
    if(count!=false){
        printf("It is a rising matrix");
    }
    else{
        printf("Not a rising Matrix");
    }
}
