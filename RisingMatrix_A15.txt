#include <stdio.h>
#include<stdbool.h>  
int main()
{
    int arr[3][3],i,j,m,n;
    bool count=true;
    printf("Enter m and n:");
    scanf("%d%d",&m,&n);
    
    for (i=0;i<m;i++)    
    {    
        for (j=0;j<n;j++)    
        {    
            printf("Enter a[%d][%d]: ",i,j);                
            scanf("%d",&arr[i][j]);    
        }    
    }    
    printf("\n printing the elements ....\n");     
    for(i=0;i<m;i++)    
    {    
        printf("\n");    
        for (j=0;j<n;j++)    
        {    
            printf("%d\t",arr[i][j]);    
        }    
    }
    
    j==0;
    for (i=0;i<m;i++)    
    {  
        for(j=0;j<n;j++){

        if(arr[j][i]>arr[j+1][i]){
            
               continue;
        }
        else{
            count=false;
        }
        }
    }
    
    if(count!=false){
        printf("It is a rising matrix");
    }
    else{
        printf("Not a rising Matrix");
    }
}
